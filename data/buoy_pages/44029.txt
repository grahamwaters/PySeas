		function requestSucceeded(response, io){

					var attr = {};
					attr["id"] = stn.id.toUpperCase();
					attr["name"] = stn.name;
					attr["data"] = stn.data;
					attr["seq"] = parseInt(stn.seq);
					if (isNaN(attr["seq"])) { attr["seq"] = null; }

					graphic.setAttributes(attr);
